-- Cerinta 4

CREATE SEQUENCE secv_id_client
  START WITH 1
  INCREMENT BY 1
  MINVALUE 1
  MAXVALUE 999999999
  NOCYCLE
  NOCACHE;

CREATE TABLE CLIENT (
  id_client INT DEFAULT secv_id_client.NEXTVAL PRIMARY KEY,
  nume VARCHAR(100) NOT NULL,
  email VARCHAR(100) UNIQUE,
  telefon VARCHAR(15) UNIQUE
);

CREATE SEQUENCE secv_id_antrenor
  START WITH 1
  INCREMENT BY 1
  MINVALUE 1
  MAXVALUE 999999999
  NOCYCLE
  NOCACHE;

CREATE TABLE ANTRENOR (
  id_antrenor INT DEFAULT secv_id_antrenor.NEXTVAL PRIMARY KEY,
  nume VARCHAR(100) NOT NULL,
  telefon VARCHAR(15) UNIQUE,
  email VARCHAR(100) UNIQUE
);


CREATE SEQUENCE secv_id_programare
  START WITH 1
  INCREMENT BY 1
  MINVALUE 1
  MAXVALUE 999999999
  NOCYCLE
  NOCACHE;

CREATE TABLE PROGRAMARE (
  id_programare INT DEFAULT secv_id_programare.NEXTVAL PRIMARY KEY,
  id_curs INT NOT NULL,
  data DATE NOT NULL,
  ora_start VARCHAR2(10) NOT NULL,
  ora_sfarsit VARCHAR2(10) NOT NULL,
  FOREIGN KEY (id_curs) REFERENCES CURS(id_curs)
);

CREATE SEQUENCE secv_id_curs
  START WITH 1
  INCREMENT BY 1
  MINVALUE 1
  MAXVALUE 999999999
  NOCYCLE
  NOCACHE;

CREATE TABLE CURS (
  id_curs INT DEFAULT secv_id_curs.NEXTVAL PRIMARY KEY,
  id_antrenor INT NOT NULL,
  id_sala INT NOT NULL,
  id_stil INT NOT NULL,
  nivel VARCHAR(20) NOT NULL CHECK (nivel IN ('începător', 'intermediar', 'avansat')),
  durata INT NOT NULL CHECK (durata > 0),
  FOREIGN KEY (id_antrenor) REFERENCES ANTRENOR(id_antrenor),
  FOREIGN KEY (id_sala) REFERENCES SALA(id_sala),
  FOREIGN KEY (id_stil) REFERENCES STILDANS(id_stil)
);

CREATE SEQUENCE secv_id_studio
  START WITH 1
  INCREMENT BY 1
  MINVALUE 1
  MAXVALUE 999999999
  NOCYCLE
  NOCACHE;

CREATE TABLE STUDIO (
  id_studio INT DEFAULT secv_id_studio.NEXTVAL PRIMARY KEY,
  nume VARCHAR(100) NOT NULL,
  oras VARCHAR(100) NOT NULL,
  adresa VARCHAR(255) NOT NULL
);

CREATE SEQUENCE secv_id_sala
  START WITH 1
  INCREMENT BY 1
  MINVALUE 1
  MAXVALUE 999999999
  NOCYCLE
  NOCACHE;

CREATE TABLE SALA (
  id_sala INT DEFAULT secv_id_sala.NEXTVAL PRIMARY KEY,
  id_studio INT NOT NULL,
  denumire VARCHAR(100) NOT NULL,
  capacitate INT NOT NULL,
  FOREIGN KEY (id_studio) REFERENCES STUDIO(id_studio)
);

CREATE SEQUENCE secv_id_stildans
  START WITH 1
  INCREMENT BY 1
  MINVALUE 1
  MAXVALUE 999999999
  NOCYCLE
  NOCACHE;

CREATE TABLE STILDANS (
  id_stil INT DEFAULT secv_id_stildans.NEXTVAL PRIMARY KEY,
  denumire VARCHAR(100) UNIQUE NOT NULL
);

CREATE SEQUENCE secv_id_sponsor
  START WITH 1
  INCREMENT BY 1
  MINVALUE 1
  MAXVALUE 999999999
  NOCYCLE
  NOCACHE;

CREATE TABLE SPONSOR (
  id_sponsor INT DEFAULT secv_id_sponsor.NEXTVAL PRIMARY KEY,
  nume_firma VARCHAR(100) NOT NULL,
  domeniu VARCHAR(100) NOT NULL
);

CREATE SEQUENCE secv_id_manager
  START WITH 1
  INCREMENT BY 1
  MINVALUE 1
  MAXVALUE 999999999
  NOCYCLE
  NOCACHE;

CREATE TABLE MANAGER (
  id_manager INT DEFAULT secv_id_manager.NEXTVAL PRIMARY KEY,
  nume VARCHAR(100) NOT NULL,
  email VARCHAR(100) UNIQUE NOT NULL,
  id_studio INT UNIQUE NOT NULL,
  FOREIGN KEY (id_studio) REFERENCES STUDIO(id_studio)
);

CREATE TABLE CLIENT_SE_INSCRIE_LA_PROGRAMARE (
  id_client INT NOT NULL,
  id_programare INT NOT NULL,
  status_plata VARCHAR(20) NOT NULL CHECK (status_plata IN ('plătit', 'neplătit', 'rezervat')),
  PRIMARY KEY (id_client, id_programare),
  FOREIGN KEY (id_client) REFERENCES CLIENT(id_client),
  FOREIGN KEY (id_programare) REFERENCES PROGRAMARE(id_programare)
);

CREATE TABLE ANTRENOR_E_COMPETENT_IN_STILDANS (
  id_antrenor INT NOT NULL,
  id_stil INT NOT NULL,
  nivel VARCHAR(20) NOT NULL CHECK (nivel IN ('începător', 'intermediar', 'avansat')),
  PRIMARY KEY (id_antrenor, id_stil),
  FOREIGN KEY (id_antrenor) REFERENCES ANTRENOR(id_antrenor),
  FOREIGN KEY (id_stil) REFERENCES STILDANS(id_stil)
);

CREATE TABLE SPONSOR_SPONSORIZEAZA_STUDIO (
  id_sponsor INT NOT NULL,
  id_studio INT NOT NULL,
  perioada VARCHAR(50) NOT NULL,
  suma NUMBER(10,2) NOT NULL,
  PRIMARY KEY (id_sponsor, id_studio, perioada),
  FOREIGN KEY (id_sponsor) REFERENCES SPONSOR(id_sponsor),
  FOREIGN KEY (id_studio) REFERENCES STUDIO(id_studio)
);

CREATE TABLE CURS_IMPLICA_ANTRENOR_IN_STILDANS (
  id_curs INT NOT NULL,
  id_antrenor INT NOT NULL,
  id_stil INT NOT NULL,
  PRIMARY KEY (id_curs, id_antrenor, id_stil),
  FOREIGN KEY (id_curs) REFERENCES CURS(id_curs),
  FOREIGN KEY (id_antrenor) REFERENCES ANTRENOR(id_antrenor),
  FOREIGN KEY (id_stil) REFERENCES STILDANS(id_stil)
);

CREATE OR REPLACE PROCEDURE raport_oferta_studio (
    p_id_studio IN NUMBER
) IS
    -- 1) 3 niveluri standard
    TYPE t_nivele_std IS VARRAY(3) OF VARCHAR2(20);
    v_nivele t_nivele_std := t_nivele_std('începător', 'intermediar', 'avansat');

    -- 2) toate cursurile din studio
    TYPE r_curs_info IS RECORD (
        id_curs NUMBER,
        nivel   VARCHAR2(20)
    );
    TYPE t_lista_cursuri IS TABLE OF r_curs_info;
    v_cursuri t_lista_cursuri;

    -- 3) cursurile pe fiecare nivel
    TYPE t_stats IS TABLE OF NUMBER INDEX BY VARCHAR2(20);
    v_statistica t_stats;

    v_nume_studio STUDIO.nume%TYPE;
    v_idx VARCHAR2(20);

BEGIN

    SELECT nume INTO v_nume_studio
    FROM STUDIO
    WHERE id_studio = p_id_studio;

    DBMS_OUTPUT.PUT_LINE('=== RAPORT OFERTA STUDIO: ' || v_nume_studio || ' ===');

    DBMS_OUTPUT.PUT_LINE('1) Niveluri standard analizate (VARRAY):');
    FOR i IN 1..v_nivele.COUNT LOOP
        DBMS_OUTPUT.PUT_LINE('   - ' || v_nivele(i));
    END LOOP;

    SELECT c.id_curs, c.nivel
    BULK COLLECT INTO v_cursuri
    FROM CURS c
    JOIN SALA s ON c.id_sala = s.id_sala
    WHERE s.id_studio = p_id_studio;

    DBMS_OUTPUT.PUT_LINE('2) Lista totală de cursuri găsite (NESTED TABLE): ' || v_cursuri.COUNT || ' cursuri.');

    v_statistica('începător') := 0;
    v_statistica('intermediar') := 0;
    v_statistica('avansat') := 0;

    IF v_cursuri.COUNT > 0 THEN
        FOR i IN v_cursuri.FIRST .. v_cursuri.LAST LOOP
            IF v_statistica.EXISTS(v_cursuri(i).nivel) THEN
                v_statistica(v_cursuri(i).nivel) := v_statistica(v_cursuri(i).nivel) + 1;
            ELSE
                v_statistica(v_cursuri(i).nivel) := 1;
            END IF;
        END LOOP;
    ELSE
        DBMS_OUTPUT.PUT_LINE('   (Nu există cursuri programate momentan)');
    END IF;

    DBMS_OUTPUT.PUT_LINE('3) Statistica pe niveluri (ASSOCIATIVE ARRAY):');
    v_idx := v_statistica.FIRST;
    WHILE v_idx IS NOT NULL LOOP
        DBMS_OUTPUT.PUT_LINE('   -> Nivel ' || v_idx || ': ' || v_statistica(v_idx) || ' cursuri');
        v_idx := v_statistica.NEXT(v_idx);
    END LOOP;

EXCEPTION
    WHEN NO_DATA_FOUND THEN
        DBMS_OUTPUT.PUT_LINE('Eroare: Nu există niciun studio cu ID-ul ' || p_id_studio);
    WHEN OTHERS THEN
        DBMS_OUTPUT.PUT_LINE('Eroare neașteptată: ' || SQLERRM);
END;
/

BEGIN
    raport_oferta_studio(3);
END;
/

CREATE OR REPLACE PROCEDURE raport_logistica_studio (
    p_id_studio IN NUMBER
) IS
    v_nume_studio STUDIO.nume%TYPE;

    -- CURSOR 1 - neparametrizat: salile
    CURSOR c_sali IS
        SELECT id_sala, denumire, capacitate
        FROM SALA
        WHERE id_studio = p_id_studio
        ORDER BY capacitate DESC;

    -- CURSOR 2 - parametrizat: cursurile
    CURSOR c_cursuri (cp_id_sala NUMBER) IS
        SELECT s.denumire AS nume_stil, c.nivel
        FROM CURS c
        JOIN STILDANS s ON c.id_stil = s.id_stil
        WHERE c.id_sala = cp_id_sala;

    v_curs_info c_cursuri%ROWTYPE;

BEGIN
    SELECT nume INTO v_nume_studio
    FROM STUDIO
    WHERE id_studio = p_id_studio;

    DBMS_OUTPUT.PUT_LINE('=== RAPORT LOGISTICA: ' || v_nume_studio || ' ===');

    FOR r_sala IN c_sali LOOP
        DBMS_OUTPUT.PUT_LINE('----------------------------------------');
        DBMS_OUTPUT.PUT_LINE('SALA: ' || r_sala.denumire || ' (Capacitate: ' || r_sala.capacitate || ' locuri)');

        OPEN c_cursuri(r_sala.id_sala);
        LOOP
            FETCH c_cursuri INTO v_curs_info;
            EXIT WHEN c_cursuri%NOTFOUND;

            DBMS_OUTPUT.PUT_LINE('   -> Curs: ' || v_curs_info.nume_stil || ' | Nivel: ' || v_curs_info.nivel);
        END LOOP;

        IF c_cursuri%ROWCOUNT = 0 THEN
             DBMS_OUTPUT.PUT_LINE('   (Aceasta sală nu are cursuri alocate)');
        END IF;

        CLOSE c_cursuri;
    END LOOP;

EXCEPTION
    WHEN NO_DATA_FOUND THEN
        DBMS_OUTPUT.PUT_LINE('Eroare: Nu există studioul cu ID-ul ' || p_id_studio);
    WHEN OTHERS THEN
        DBMS_OUTPUT.PUT_LINE('Eroare neașteptată: ' || SQLERRM);
END;
/

BEGIN
    raport_logistica_studio(2);
END;
/

CREATE OR REPLACE FUNCTION get_data_programare_client(
    p_id_client IN NUMBER
) RETURN VARCHAR2
IS
    v_nume_client CLIENT.nume%TYPE;
    v_data_prog   DATE;
BEGIN
    SELECT c.nume, p."DATA"
    INTO v_nume_client, v_data_prog
    FROM CLIENT c
    JOIN CLIENT_SE_INSCRIE_LA_PROGRAMARE i ON c.id_client = i.id_client
    JOIN PROGRAMARE p ON i.id_programare = p.id_programare
    WHERE c.id_client = p_id_client
      AND p."DATA" >= TRUNC(SYSDATE);

    RETURN 'Clientul ' || v_nume_client || ' are o programare viitoare pe data de: ' || TO_CHAR(v_data_prog, 'DD-MM-YYYY');

EXCEPTION
    WHEN NO_DATA_FOUND THEN
        RETURN 'CLIENTUL NU ARE PROGRAMARI VIITOARE';

    WHEN TOO_MANY_ROWS THEN
        RETURN 'CLIENTUL ARE MAI MULTE PROGRAMARI VIITOARE';
END;
/

CREATE OR REPLACE PROCEDURE verifica_relatie_client_antr(
    p_id_client   IN NUMBER,
    p_id_antrenor IN NUMBER
) IS
    e_client_inexistent   EXCEPTION;
    e_antrenor_inexistent EXCEPTION;
    e_fara_interactiune   EXCEPTION;

    v_nume_client   CLIENT.nume%TYPE;
    v_nume_antrenor ANTRENOR.nume%TYPE;
    v_nr_intalniri  NUMBER;

BEGIN
    SELECT
        (SELECT nume FROM CLIENT WHERE id_client = p_id_client),

        (SELECT nume FROM ANTRENOR WHERE id_antrenor = p_id_antrenor),

        (SELECT COUNT(*)
         FROM CLIENT_SE_INSCRIE_LA_PROGRAMARE inr
         JOIN PROGRAMARE p ON inr.id_programare = p.id_programare
         JOIN CURS c ON p.id_curs = c.id_curs
         WHERE inr.id_client = p_id_client
           AND c.id_antrenor = p_id_antrenor)

    INTO v_nume_client, v_nume_antrenor, v_nr_intalniri
    FROM DUAL;

    IF v_nume_client IS NULL THEN
        RAISE e_client_inexistent;
    END IF;

    IF v_nume_antrenor IS NULL THEN
        RAISE e_antrenor_inexistent;
    END IF;

    IF v_nr_intalniri = 0 THEN
        RAISE e_fara_interactiune;
    END IF;

    DBMS_OUTPUT.PUT_LINE('SUCCES: Clientul ' || v_nume_client ||
                         ' a participat la ' || v_nr_intalniri ||
                         ' sesiuni cu antrenorul ' || v_nume_antrenor || '.');

EXCEPTION
    WHEN e_client_inexistent THEN
        DBMS_OUTPUT.PUT_LINE('EROARE: Nu exista client cu ID-ul ' || p_id_client);

    WHEN e_antrenor_inexistent THEN
        DBMS_OUTPUT.PUT_LINE('EROARE: Nu exista antrenor cu ID-ul ' || p_id_antrenor);

    WHEN e_fara_interactiune THEN
        DBMS_OUTPUT.PUT_LINE('INFO: Clientul ' || v_nume_client ||
                             ' nu a fost niciodata la antrenorul ' || v_nume_antrenor);

    WHEN OTHERS THEN
        DBMS_OUTPUT.PUT_LINE('Eroare neasteptata: ' || SQLERRM);
END;
/

SET SERVEROUTPUT ON;
BEGIN
    DBMS_OUTPUT.PUT_LINE('--- TEST 1: SUCCES ---');
    verifica_relatie_client_antr(10, 20);

    DBMS_OUTPUT.PUT_LINE('----------------------');

    DBMS_OUTPUT.PUT_LINE('--- TEST 2: CLIENT INEXISTENT ---');
    verifica_relatie_client_antr(999, 1);

    DBMS_OUTPUT.PUT_LINE('----------------------');

    DBMS_OUTPUT.PUT_LINE('--- TEST 3: ANTRENOR INEXISTENT ---');
    verifica_relatie_client_antr(1, 999);

    DBMS_OUTPUT.PUT_LINE('----------------------');

    DBMS_OUTPUT.PUT_LINE('--- TEST 4: FARA INTERACTIUNE ---');
    verifica_relatie_client_antr(1, 2);
END;
/

CREATE OR REPLACE TRIGGER trg_validare_data_prog
BEFORE INSERT OR UPDATE ON PROGRAMARE
FOR EACH ROW
BEGIN
    IF :NEW."DATA" < TRUNC(SYSDATE) THEN
        RAISE_APPLICATION_ERROR(-20001, 'EROARE: Nu se pot face programări în trecut!');
    END IF;
END;
/
